CREATE TABLE DEPARTMENTS(
DEPT_ID NUMBER(4) PRIMARY KEY,
DEPT_NAME VARCHAR(20) UNIQUE,
HEAD_OF_DEPARTMENTS VARCHAR(20));


CREATE TABLE EMPLOYEE(
DEPT_ID NUMBER(4) REFERENCES DEPARTMENTS(DEPT_ID) ON DELETE CASCADE,
EMP_ID NUMBER(4) PRIMARY KEY,
EMP_NAME VARCHAR(20),
EMP_PHONE NUMBER(10) UNIQUE,
EMP_MAIL VARCHAR(50) UNIQUE,
EMP_ADDRESS VARCHAR(20));


CREATE TABLE PAYROLL(
DEPT_ID NUMBER(4) REFERENCES DEPARTMENTS(DEPT_ID) ON DELETE CASCADE,
EMP_ID NUMBER(4) REFERENCES EMPLOYEE(EMP_ID) ON DELETE CASCADE,
BASIC NUMBER(20),
BONUS NUMBER(10),
HRA NUMBER(10),
MEDICAL_ALLOWANCE NUMBER(10),
TA NUMBER(10),
DA NUMBER(10),
TAX NUMBER(10),
LEAVE NUMBER(4),
NET_SALARY NUMBER(20));
